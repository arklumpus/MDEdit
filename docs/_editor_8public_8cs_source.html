213  /// <summary>
214  /// Create a new <see cref="Editor"/> instance.
215  /// </summary>
216  /// <param name="initialText">The initial text of the editor.</param>
217  /// <param name="guid">A unique identifier for the document being edited. If this is <see langword="null"/>, a new <see cref="System.Guid"/> is generated. If the same identifier is used multiple times, the save history of the document will be available, even if the application has been closed between different sessions.</param>
218  /// <param name="additionalShortcuts">Additional application-specific shortcuts (for display purposes only - you need to implement your own logic).</param>
219  /// <returns>A fully initialised <see cref="Editor"/> instance.</returns>
220  public static async Task<Editor> Create(string initialText = "", string guid = null, Shortcut[] additionalShortcuts = null)
221  {
222  if (string.IsNullOrEmpty(guid))
223  {
224  guid = System.Guid.NewGuid().ToString("N");
225  }
226  else
227  {
228  foreach (char c in Path.GetInvalidPathChars().Concat(Path.GetInvalidFileNameChars()))
229  {
230  if (guid.Contains(c))
231  {
232  throw new ArgumentException("The provided Guid \"" + guid + "\" is not valid!\nThe Guid must be a valid identifier for a path or a file.", nameof(guid));
233  }
234  }
235  }
236 
237  AsynchronousImageCache.SetExitEventHandler();
238 
239  Editor tbr = new Editor(false);
240  await tbr.Initialize(initialText, guid, additionalShortcuts ?? new Shortcut[0]);
241  return tbr;
242  }

264  /// <summary>
265  /// Add the current text of the document to the save history (if enabled) and invoke the <see cref="SaveRequested"/> event.
266  /// </summary>
267  public void Save()
268  {
269  string text = this.EditorControl.Text.ToString();
270 
271  if (KeepSaveHistory)
272  {
273  string autosaveDirectory = Path.Combine(Environment.GetFolderPath(Environment.SpecialFolder.LocalApplicationData), Assembly.GetEntryAssembly().GetName().Name);
274  Directory.CreateDirectory(Path.Combine(autosaveDirectory, this.Guid));
275  System.IO.File.WriteAllText(System.IO.Path.Combine(autosaveDirectory, this.Guid, DateTimeOffset.UtcNow.ToUnixTimeSeconds().ToString() + ".md"), text);
276  }
277 
278  this.UpdateLastSavedDocument();
279  if (this.SaveHistoryContainer.IsVisible)
280  {
281  this.SaveHistoryContainer.Refresh();
282  }
283 
284  this.InvokeSaveRequested(new SaveEventArgs(text));
285  }
286  }
