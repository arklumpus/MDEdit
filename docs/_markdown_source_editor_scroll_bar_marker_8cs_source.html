19 using Avalonia.Controls;
20 using Avalonia.Media;
21 using Microsoft.CodeAnalysis.Text;
22 using System.Collections.Generic;
23 using System.Linq;
24 
25 namespace MDEdit
26 {
27  internal class MarkdownSourceEditorScrollBarMarker : Control
28  {
29  private MarkdownSourceEditor Editor;
30 
31  public MarkdownSourceEditorScrollBarMarker(MarkdownSourceEditor editor)
32  {
33  this.Editor = editor;
34  this.Width = 18;
35  this.HorizontalAlignment = Avalonia.Layout.HorizontalAlignment.Right;
36  this.IsHitTestVisible = false;
37  }
38 
39  private static IBrush CaretPositionBrush = new SolidColorBrush(Color.FromRgb(0, 0, 205));
40  private static SolidColorBrush SearchBrush = new SolidColorBrush(Color.FromRgb(246, 185, 127));
41 
42  public override void Render(DrawingContext context)
43  {
44  base.Render(context);
45 
46  if (Editor.ShowScrollbarOverview && Editor.Extent.Height > Editor.Bounds.Height)
47  {
48  double totalHeight = this.Bounds.Height - 36;
49 
50  if (Editor.Extent.Width > Editor.Bounds.Width)
51  {
52  totalHeight -= 18;
53  }
54 
55  double totalLines = Editor.Text.Lines.Count;
56 
57  foreach (HighlightedLineRange range in this.Editor.HighlightedLines)
58  {
59  context.FillRectangle(range.HighlightBrush, MarkdownSourceEditorCaret.Round(new Avalonia.Rect(0, 18 + range.LineSpan.Start / totalLines * totalHeight, 4, (range.LineSpan.Length + 1) / totalLines * totalHeight), new Avalonia.Size(1, 1)));
60  }
61 
62  HashSet<int> greenLines = new HashSet<int>();
63 
64  for (int i = 0; i < this.Editor.Markers.Count; i++)
65  {
66  LinePositionSpan markerSpan = this.Editor.Text.Lines.GetLinePositionSpan(this.Editor.Markers[i].Span);
67  foreach (int j in Enumerable.Range(markerSpan.Start.Line, markerSpan.End.Line - markerSpan.Start.Line + 1))
68  {
69  greenLines.Add(j);
70  }
71  }
72 
73  List<int> sortedGreenLines = new List<int>(greenLines);
74 
75  sortedGreenLines.Sort();
76 
77  IEnumerable<HighlightedLineRange> greenLineSpans = from el in (from el in sortedGreenLines select new TextSpan(el, 0)).Join() select new HighlightedLineRange(el, Brushes.Green);
78 
79  foreach (HighlightedLineRange range in greenLineSpans)
80  {
81  context.FillRectangle(range.HighlightBrush, MarkdownSourceEditorCaret.Round(new Avalonia.Rect(14, 18 + range.LineSpan.Start / totalLines * totalHeight, 4, (range.LineSpan.Length + 1) / totalLines * totalHeight), new Avalonia.Size(1, 1)));
82  }
83 
84  if (this.Editor.SearchReplace.IsVisible)
85  {
86  HashSet<int> searchMatches = new HashSet<int>();
87 
88  foreach (SearchSpan span in this.Editor.SearchSpans)
89  {
90  LinePositionSpan lineSpan = this.Editor.Text.Lines.GetLinePositionSpan(span.Span);
91  foreach (int j in Enumerable.Range(lineSpan.Start.Line, lineSpan.End.Line - lineSpan.Start.Line + 1))
92  {
93  searchMatches.Add(j);
94  }
95  }
96 
97  foreach (int line in searchMatches)
98  {
99  context.FillRectangle(SearchBrush, MarkdownSourceEditorCaret.Round(new Avalonia.Rect(7, 18 + line / totalLines * totalHeight, 4, 1.0 / totalLines * totalHeight), new Avalonia.Size(1, 1)));
100  }
101  }
102 
103  context.FillRectangle(CaretPositionBrush, MarkdownSourceEditorCaret.Round(new Avalonia.Rect(0, 18 + (Editor.Text.Lines.GetLinePosition(Editor.CaretOffset).Line + 0.5) / totalLines * totalHeight - 1, 18, 2), new Avalonia.Size(1, 1)));
104  }
105  }
106  }
107 }
