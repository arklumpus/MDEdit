94  /// <summary>
95  /// Resolves an image Uri asynchronously. If the image has already been downloaded previously and is available in the cache, its path on disk is returned immediately.
96  /// Otherwise, it is queued for download and when it becomes available, the <see cref="CacheUpdated"/> event is invoked. If an image is requested again while it is
97  /// being downloaded, a new download of the image is prevented.
98  /// </summary>
99  /// <param name="imageUri">The Uri of the image to download, either absolute or relative to the <paramref name="baseUriString"/>.</param>
100  /// <param name="baseUriString">The base Uri for resolving the <paramref name="imageUri"/>.</param>
101  /// <returns>An <see cref="ImageRetrievalResult"/> containing the path to a copy of the image file on disk. If the <see cref="ImageRetrievalResult.ImagePath"/>
102  /// property of the return value is <see langword="null"/>, the image was not available in the cache and has been queued for download.</returns>
103  public static ImageRetrievalResult ImageUriResolverAsynchronous(string imageUri, string baseUriString)
104  {
105  bool found;
106  string cachedImage;
107 
108  lock (cacheLock)
109  {
110  found = imageCache.TryGetValue(baseUriString + "|||" + imageUri, out cachedImage);
111  }
112 
113  if (!found)
114  {
115  bool notAlreadyFetching;
116 
117  lock (cacheLock)
118  {
119  notAlreadyFetching = fetchingImages.Add(baseUriString + "|||" + imageUri);
120  }
121 
122  if (notAlreadyFetching)
123  {
124  Thread thr = new Thread(() =>
125  {
126  (string imagePath, bool wasDownloaded) = VectSharp.Markdown.HTTPUtils.ResolveImageURI(imageUri, baseUriString);
127 
128  if (!string.IsNullOrEmpty(imagePath) && File.Exists(imagePath))
129  {
130  string id = Guid.NewGuid().ToString();
131 
132  string cachedImage = Path.Combine(imageCacheFolder, id + Path.GetExtension(imagePath));
133 
134  if (wasDownloaded)
135  {
136  if (!Directory.Exists(imageCacheFolder))
137  {
138  Directory.CreateDirectory(imageCacheFolder);
139  }
140 
141  File.Move(imagePath, cachedImage);
142  Directory.Delete(Path.GetDirectoryName(imagePath));
143  }
144  else
145  {
146  File.Copy(imagePath, cachedImage);
147  }
148 
149  lock (cacheLock)
150  {
151  imageCache[baseUriString + "|||" + imageUri] = cachedImage;
152  fetchingImages.Remove(baseUriString + "|||" + imageUri);
153  }
154 
155  CacheUpdated?.Invoke(null, new CacheUpdatedEventArgs(baseUriString, imageUri));
156  }
157  else
158  {
159  lock (cacheLock)
160  {
161  fetchingImages.Remove(baseUriString + "|||" + imageUri);
162  }
163  }
164  });
165 
166  thr.Start();
167  }
168 
169  return new ImageRetrievalResult(null, false);
170  }
171  else
172  {
173  return new ImageRetrievalResult(cachedImage, false);
174  }
175  }

177  /// <summary>
178  /// Resolves an image Uri synchronously. If the image has already been downloaded previously and is available in the cache, its path on disk is returned immediately.
179  /// Otherwise, this method blocks until the image is downloaded. If an image is requested by this method while it is being downloaded as a result of a call to
180  /// <see cref="ImageUriResolverAsynchronous(string, string)"/>, it is downloaded a second time.
181  /// </summary>
182  /// <param name="imageUri">The Uri of the image to download, either absolute or relative to the <paramref name="baseUriString"/>.</param>
183  /// <param name="baseUriString">The base Uri for resolving the <paramref name="imageUri"/>.</param>
184  /// <returns>An <see cref="ImageRetrievalResult"/> containing the path to a copy of the image file on disk. If the <see cref="ImageRetrievalResult.ImagePath"/>
185  /// property of the return value is <see langword="null"/>, an error occurred while the image was being accessed (e.g. it was not found on the server).</returns>
186  public static ImageRetrievalResult ImageUriResolverSynchronous(string imageUri, string baseUriString)
187  {
188  lock (cacheLock)
189  {
190  if (!imageCache.TryGetValue(baseUriString + "|||" + imageUri, out string cachedImage))
191  {
192  (string imagePath, bool wasDownloaded) = VectSharp.Markdown.HTTPUtils.ResolveImageURI(imageUri, baseUriString);
193 
194  if (!string.IsNullOrEmpty(imagePath) && File.Exists(imagePath))
195  {
196  string id = Guid.NewGuid().ToString();
197 
198  cachedImage = Path.Combine(imageCacheFolder, id + Path.GetExtension(imagePath));
199 
200  if (wasDownloaded)
201  {
202  if (!Directory.Exists(imageCacheFolder))
203  {
204  Directory.CreateDirectory(imageCacheFolder);
205  }
206 
207  File.Move(imagePath, cachedImage);
208  Directory.Delete(Path.GetDirectoryName(imagePath));
209  }
210  else
211  {
212  File.Copy(imagePath, cachedImage);
213  }
214 
215  imageCache[baseUriString + "|||" + imageUri] = cachedImage;
216  }
217  else
218  {
219  cachedImage = null;
220  }
221  }
222 
223  return new ImageRetrievalResult(cachedImage, false);
224  }
225  }
